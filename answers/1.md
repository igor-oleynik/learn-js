```js
function getType(num) {
    switch(typeof num) {
        case 'number':
        case 'bigInt':
            return 'number';

        case 'boolean':
            return (num === true ? 'true' : num === false ? 'false' : '');

        case 'string':
            return 'string';
            
            // case 'object':
        case null:
            return 'null';

        default:
            return 'else';
    }
}
```
